[docs]    @staticmethod
    def bound_references(struct, exprs, case_sensitive=True):
        if exprs is None:
            return set()
        visitor = Binder.ReferenceVisitor()
        for expr in exprs:
            ExpressionVisitors.visit(Binder.bind(struct, expr, case_sensitive), visitor)

        return visitor.references
